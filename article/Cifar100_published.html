# ==== ARQUITETURA LeNet-5 ADAPTADA PARA CIFAR-100 ====
# LeNet-5 adaptada: in_channels=3 (RGB), out_features=100 (100 classes)

# Definio da arquitetura usando nn.Sequential para simplificar
model = nn.Sequential(
    # === BLOCO CONVOLUCIONAL 1 ===
    # Conv2d(in_channels=3, out_channels=6, kernel_size=5)
    # - Entrada: imagens RGB 32x32x3
    # - Sada: 6 feature maps de 28x28 (32-5+1=28)
    # - Detecta caractersticas bsicas (bordas, cores, texturas)
    nn.Conv2d(3, 6, 5),

    # ReLU: funo de ativao no-linear
    # - Transforma valores negativos em zero
    # - Permite que a rede aprenda padres complexos
    nn.ReLU(),

    # MaxPool2d(kernel_size=2, stride=2)
    # - Reduz dimensionalidade de 28x28 para 14x14
    # - Mantm as caractersticas mais importantes de cada regio
    # - Torna a rede mais eficiente e robusta a pequenas variaes
    nn.MaxPool2d(2, 2),

    # === BLOCO CONVOLUCIONAL 2 ===
    # Conv2d(in_channels=6, out_channels=16, kernel_size=5)
    # - Entrada: 6 feature maps de 14x14
    # - Sada: 16 feature maps de 10x10 (14-5+1=10)
    # - Combina features do bloco anterior para detectar padres mais complexos
    nn.Conv2d(6, 16, 5),

    # ReLU: ativao no-linear
    nn.ReLU(),

    # MaxPool2d: reduz de 10x10 para 5x5
    # - Consolida representaes antes da classificao
    nn.MaxPool2d(2, 2),

    # === BLOCO CLASSIFICADOR ===
    # Flatten: transforma tensor 16x5x5 em vetor 1D de 400 elementos
    # - Prepara dados para as camadas fully connected
    nn.Flatten(),

    # Linear(in_features=400, out_features=120)
    # - Primeira camada densa: 16*5*5=400 -> 120
    # - Aprende combinaes complexas das features extradas
    nn.Linear(16*5*5, 120),
    nn.ReLU(),

    # Linear(120, 84): segunda camada densa
    # - Refina as representaes aprendidas
    nn.Linear(120, 84),
    nn.ReLU(),

    # Linear(84, 100): camada de sada
    # - 100 neurnios, um para cada classe do CIFAR-100
    # - Os logits so convertidos em probabilidades pela CrossEntropyLoss
    nn.Linear(84, 100)
)

# Mover modelo para GPU se disponvel
model = model.to(device)

def calc_ewma(x, alpha): """Calcula a mdia mvel exponencial (EWMA) simples sobre um vetor 1D.

Args:
    x: array-like 1D
    alpha: fator de atualizao (0 < alpha <= 1). Valores menores -> suavizao maior.

Returns:
    np.ndarray: EWMA do mesmo tamanho de `x`.
"""
import numpy as _np
x = _np.asarray(x, dtype=float)
y = _np.empty_like(x)
if x.size == 0:
    return y
y[0] = x[0]
for t in range(1, x.size):
    y[t] = alpha * x[t] + (1 - alpha) * y[t-1]
return y

def calc_corrected_ewma(x, alpha): """Calcula EWMA com correo de vis (como em Adam).

Interpreta `alpha` como coeficiente de atualizao e aplica correo de vis
retornando o vetor m_hat (m_t / (1 - beta^t)).
"""
import numpy as _np
x = _np.asarray(x, dtype=float)
m = _np.zeros_like(x)
m_hat = _np.zeros_like(x)
beta = 1 - alpha
if x.size == 0:
    return m_hat
m[0] = x[0]
m_hat[0] = m[0] / (1 - beta**1)
for t in range(1, x.size):
    m[t] = alpha * x[t] + (1 - alpha) * m[t-1]
    m_hat[t] = m[t] / (1 - beta**(t+1))
return m_hat

EWMA (Mdia Mvel Exponencial)

As funes acima implementam utilitrios prticos para anlise de gradientes:

  • calc_ewma(x, alpha): mdia mvel exponencial simples usada para suavizar sinais ruidosos (ex.: gradientes).
  • calc_corrected_ewma(x, alpha): EWMA com correo de vis, equivalente correo usada em Adam (til para comparar estatsticas no viciadas).

Use esses utilitrios nas prximas clulas para gerar comparaes entre gradientes crus, gradientes suavizados por EWMA e a verso corrigida (til para entender o efeito de / em otimizadores adaptativos).

for epoch in range(num_epochs):
    # === FASE DE TREINO ===
    model.train()  # Coloca modelo em modo de treinamento
    running_loss = 0.0
    correct = 0
    total = 0

    # Iterar sobre batches de treino
    for i, (inputs, labels) in enumerate(train_loader):
        # Mover dados para GPU se disponvel
        inputs, labels = inputs.to(device), labels.to(device)

        # Zerar gradientes acumulados da iterao anterior
        optimizer.zero_grad()

        # Forward pass: calcular predies do modelo
        outputs = model(inputs)

        # Calcular loss (diferena entre predio e label real)
        loss = criterion(outputs, labels)

        # Backward pass: calcular gradientes
        loss.backward()

        # Atualizar pesos do modelo usando gradientes
        optimizer.step()

        # Acumular estatsticas para clculo de mtricas
        running_loss += loss.item()
        _, predicted = torch.max(outputs.data, 1)
        total += labels.size(0)
        correct += (predicted == labels).sum().item()

    # Calcular mtricas de treino
    train_loss = running_loss / len(train_loader)
    train_acc = 100 * correct / total
    train_losses.append(train_loss)
    train_accuracies.append(train_acc)

    # === FASE DE VALIDAO ===
    model.eval()  # Coloca modelo em modo de avaliao (desabilita dropout, etc)
    test_loss = 0.0
    correct = 0
    total = 0

    # Desabilitar clculo de gradientes (economiza memria e acelera)
    with torch.no_grad():

    test_losses.append(test_loss)

    # Exibir progresso a cada poca

# Hooks permitem capturar ativaes intermedirias das camadas

# Dicionrio para armazenar ativaes
activations = {}

# Funo hook que ser chamada durante forward pass
def get_activation(name):
    def hook(model, input, output):
        # Armazena a sada (ativao) da camada
        activations[name] = output.detach()
    return hook

# Registrar hooks em todas as camadas convolucionais e de pooling

# Demo curto comparando Adam vs SGD (momentum)
import torch
import torch.nn as nn
import torch.optim as optim
import torchvision
import torchvision.transforms as transforms
from torch.utils.data import DataLoader
import matplotlib.pyplot as plt
import time

device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')

# LeNet-like (redefinido localmente para garantir independncia)
class LeNetLike(nn.Module):
    def __init__(self, num_classes=100):
        super().__init__()
        self.features = nn.Sequential(
            nn.Conv2d(3, 6, 5, padding=2),
            nn.ReLU(),
            nn.MaxPool2d(2),
            nn.Conv2d(6, 16, 5),
            nn.ReLU(),
            nn.MaxPool2d(2),
        )
        self.classifier = nn.Sequential(
            nn.Flatten(),
            nn.Linear(16 * 6 * 6, 120),
            nn.ReLU(),
            nn.Linear(120, 84),
            nn.ReLU(),
            nn.Linear(84, num_classes),
        )
    def forward(self, x):
        x = self.features(x)
        x = self.classifier(x)
        return x

# Data (small demo): limit batches per run to keep quick
transform = transforms.Compose([
    transforms.ToTensor(),
    transforms.Normalize((0.5071, 0.4867, 0.4408), (0.2675, 0.2565, 0.2761)),
])

trainset = torchvision.datasets.CIFAR100(root='./data', train=True, download=False, transform=transform)
train_loader = DataLoader(trainset, batch_size=128, shuffle=True, num_workers=2)

criterion = nn.CrossEntropyLoss()

def short_run(optim_name, lr, max_batches=30, epochs=3):
    torch.manual_seed(0)
    model = LeNetLike(num_classes=100).to(device)
    if optim_name == 'adam':
        optimizer = optim.Adam(model.parameters(), lr=lr)
    else:
        optimizer = optim.SGD(model.parameters(), lr=lr, momentum=0.9)

    losses = []
    it = 0
    start = time.time()
    for epoch in range(epochs):
        for batch_idx, (inputs, targets) in enumerate(train_loader):
            inputs, targets = inputs.to(device), targets.to(device)
            optimizer.zero_grad()
            outputs = model(inputs)
            loss = criterion(outputs, targets)
            loss.backward()
            optimizer.step()
            losses.append(loss.item())
            it += 1
            if it >= max_batches:
                break
        if it >= max_batches:
            break
    elapsed = time.time() - start
    return losses, elapsed

# Run short comparisons
adam_losses, adam_time = short_run('adam', lr=1e-3, max_batches=40, epochs=3)
sgd_losses, sgd_time = short_run('sgd', lr=1e-2, max_batches=40, epochs=3)

def lr_range_test(model_fn, optimizer_cls, loader, device, start_lr=1e-7, end_lr=1, num_iters=60):
    model = model_fn().to(device)
    optimizer = optimizer_cls(model.parameters(), lr=start_lr)
    mult = (end_lr / start_lr) ** (1.0 / num_iters)
    lr = start_lr
    lrs = []
    losses = []
    it = 0
    model.train()
    for inputs, targets in loader:
        inputs, targets = inputs.to(device), targets.to(device)
        optimizer.zero_grad()
        outputs = model(inputs)
        loss = criterion(outputs, targets)
        loss.backward()
        optimizer.step()
        lrs.append(lr)
        losses.append(loss.item())
        lr *= mult
        for g in optimizer.param_groups:
            g['lr'] = lr
        it += 1
        if it >= num_iters:
            break
    return lrs, losses

# ==== CAPTURA DE GRADIENTES: BRUTOS, EWMA E ADAPTADOS (Adam) ====
import matplotlib.pyplot as plt
import numpy as np

# Funo utilitria para coletar gradientes e estados do otimizador
def capture_gradients(model, optimizer, criterion, inputs, labels, device, alpha=0.1, eps=1e-8):
    model.train()
    inputs, labels = inputs.to(device), labels.to(device)
    optimizer.zero_grad()
    outputs = model(inputs)
    loss = criterion(outputs, labels)
    loss.backward()

    grads = {}
    ewmas = {}
    adapted = {}

    # Iterar sobre parmetros e coletar gradientes (flattened)
    for name, p in model.named_parameters():
        if p.grad is None:
            continue
        g = p.grad.detach().cpu().numpy().ravel()
        grads[name] = g.copy()
        # EWMA sobre o vetor de gradientes (usando mdia mvel simples ao longo do vetor)
        # Aqui aplicamos EWMA 1D no vetor para visualizao
        def vec_ewma(x, alpha):
            y = np.empty_like(x)
            if x.size==0: return y
            y[0] = x[0]
            for t in range(1, x.size):
                y[t] = alpha * x[t] + (1-alpha) * y[t-1]
            return y
        ewmas[name] = vec_ewma(g, alpha)

        # Se o otimizador for Adam, tentar recuperar exp_avg e exp_avg_sq
        adapted_vec = None
        for group in optimizer.param_groups:
            break
        state = optimizer.state.get(p, {})
        if 'exp_avg' in state and 'exp_avg_sq' in state:
            exp_avg = state['exp_avg'].detach().cpu().numpy().ravel()
            exp_avg_sq = state['exp_avg_sq'].detach().cpu().numpy().ravel()
            adapted_vec = exp_avg / (np.sqrt(exp_avg_sq) + eps)
        elif hasattr(p, 'grad'):
            adapted_vec = g.copy()
        adapted[name] = adapted_vec

    return grads, ewmas, adapted, loss.item()

Explicao rpida: Adam parmetros e correo de vis

O otimizador Adam combina duas ideias principais: (i) momentum exponencial sobre os gradientes (exp_avg) e (ii) escalonamento por uma mdia exponencial dos quadrados dos gradientes (exp_avg_sq).

  • (beta1): coeficiente de decaimento para o primeiro momento (exp_avg). Valores tpicos: 0.9 controla quanto do histrico recente mantido como momentum.
  • (beta2): coeficiente de decaimento para o segundo momento (exp_avg_sq). Valores tpicos: 0.999 controla a mdia dos gradientes ao quadrado (escala/variao).
  • (eps): pequeno termo numrico adicionado na diviso para estabilidade numrica (evita diviso por zero).

Correo de vis: durante as primeiras iteraes os termos exp_avg e exp_avg_sq esto viciados para valores prximos de zero (porque iniciam em zero). Adam aplica correes do tipo m_hat = exp_avg / (1 - ^t) e v_hat = exp_avg_sq / (1 - ^t) para obter estimativas no viciadas, o que melhora a estabilidade inicial do algoritmo.

No notebook usamos demonstraes curtas para ilustrar comportamento (curvas de perda e magnitudes de gradientes). Para a nota tcnica, inclua tambm um pargrafo curto com referncias (ex.: Kingma & Ba, 2015).
